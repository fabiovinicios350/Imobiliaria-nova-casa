CREATE DATABASE imobiliaria;
USE imobiliaria;
CREATE TABLE imoveis(
codigo INT AUTO_INCREMENT NOT NULL,
nome VARCHAR(30), 
cnpj  VARCHAR(18) , 
email  VARCHAR(50) , 
telefone  VARCHAR(13), 
endereco  VARCHAR(30),
numero CHAR(4),
cep  VARCHAR(9), 
bairro  VARCHAR(45) ,
cidade  VARCHAR(45),
complemento VARCHAR(20), 
tipoimovel VARCHAR(13) ,
tipodevenda  VARCHAR(8), 
dormitorios VARCHAR (2) ,
suites VARCHAR (2) ,
vagas VARCHAR (2) ,
fotoprincipal VARCHAR(100), 
foto1 VARCHAR(100),
foto2 VARCHAR(100),
foto3 VARCHAR(100),
foto4 VARCHAR(100),
foto5 VARCHAR(100),
descricao VARCHAR(250),
quarto VARCHAR (2) ,
banheiro VARCHAR (2) ,
cozinhas VARCHAR (2) ,
salas VARCHAR (2) ,
lavanderia VARCHAR (2) ,
porao VARCHAR (2) ,
area VARCHAR (2) ,
piscina VARCHAR (2) ,
valorvenda VARCHAR(14),
tipopagamento VARCHAR(14),
valoraluguel VARCHAR(14),
PRIMARY KEY(codigo)
);
